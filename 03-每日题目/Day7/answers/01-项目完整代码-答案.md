## 完整代码

```python
"""
命令行通讯录管理器
- 添加联系人（姓名、电话、邮箱）
- 查看所有联系人
- 搜索联系人
- 删除联系人
- 自动保存到 JSON 文件
"""
import json
import os

# 文件名常量
FILENAME = "contacts.json"

# ========== 1. 数据加载与保存 ==========

def load_contacts():
    """从 JSON 文件加载联系人列表，文件不存在时返回空列表"""
    try:
        with open(FILENAME, "r", encoding="utf-8") as f:
            return json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        # 文件不存在或内容损坏 → 返回空列表
        return []


def save_contacts(contacts):
    """将联系人列表保存到 JSON 文件"""
    with open(FILENAME, "w", encoding="utf-8") as f:
        json.dump(contacts, f, ensure_ascii=False, indent=2)
    print(f"✅ 已保存 {len(contacts)} 个联系人到 {FILENAME}")


# ========== 2. 核心功能函数 ==========

def add_contact(contacts):
    """添加新联系人"""
    print("\n--- 添加联系人 ---")
    name = input("姓名：").strip()
    phone = input("电话：").strip()
    email = input("邮箱：").strip()

    # 简单校验：姓名不能为空
    if not name:
        print("❌ 姓名不能为空！")
        return

    # 每个联系人用 dict 存储
    contact = {"name": name, "phone": phone, "email": email}
    contacts.append(contact)
    print(f"✅ 已添加联系人：{name}")


def view_all(contacts):
    """查看所有联系人"""
    print("\n--- 全部联系人 ---")
    if not contacts:
        print("📭 通讯录为空")
        return

    # 用 f-string 格式化打印表格
    print(f"{'序号':<5} {'姓名':<12} {'电话':<15} {'邮箱':<20}")
    print("-" * 52)
    for i, c in enumerate(contacts, start=1):
        print(f"{i:<5} {c['name']:<12} {c['phone']:<15} {c['email']:<20}")
    print(f"\n共 {len(contacts)} 个联系人")


def search_contact(contacts):
    """按姓名搜索联系人（支持模糊匹配）"""
    print("\n--- 搜索联系人 ---")
    keyword = input("请输入姓名关键词：").strip()

    # 遍历查找，用 in 做模糊匹配
    results = []
    for c in contacts:
        if keyword.lower() in c["name"].lower():
            results.append(c)

    if not results:
        print(f"❌ 未找到包含「{keyword}」的联系人")
        return

    print(f"\n找到 {len(results)} 个匹配：")
    for c in results:
        print(f"  姓名：{c['name']}  |  电话：{c['phone']}  |  邮箱：{c['email']}")


def delete_contact(contacts):
    """删除联系人"""
    print("\n--- 删除联系人 ---")
    name = input("请输入要删除的联系人姓名：").strip()

    # 先找到匹配的索引
    target_index = None
    for i, c in enumerate(contacts):
        if c["name"].lower() == name.lower():
            target_index = i
            break

    if target_index is None:
        print(f"❌ 未找到联系人：{name}")
        return

    # 确认后删除
    c = contacts[target_index]
    confirm = input(f"确定删除 {c['name']}（{c['phone']}）？(y/n)：").strip().lower()
    if confirm == "y":
        removed = contacts.pop(target_index)
        print(f"✅ 已删除联系人：{removed['name']}")
    else:
        print("已取消")


# ========== 3. 主菜单 ==========

def show_menu():
    """显示菜单"""
    print("\n" + "=" * 36)
    print("   📇 通讯录管理器")
    print("=" * 36)
    print("   1. 添加联系人")
    print("   2. 查看全部联系人")
    print("   3. 搜索联系人")
    print("   4. 删除联系人")
    print("   0. 退出并保存")
    print("=" * 36)


def main():
    """主函数：程序入口"""
    # 启动时加载已有数据
    contacts = load_contacts()
    if contacts:
        print(f"📂 已加载 {len(contacts)} 个联系人")

    # 主循环
    while True:
        show_menu()
        choice = input("请选择操作（0-4）：").strip()

        if choice == "1":
            add_contact(contacts)
        elif choice == "2":
            view_all(contacts)
        elif choice == "3":
            search_contact(contacts)
        elif choice == "4":
            delete_contact(contacts)
        elif choice == "0":
            save_contacts(contacts)
            print("👋 再见！")
            break
        else:
            print("❌ 无效选项，请输入 0-4")


# 程序入口
if __name__ == "__main__":
    main()
```

## 知识点总结

| 知识点 | 应用位置 |
|--------|----------|
| `list` + `dict` | 数据存储：`contacts = [{"name": ..., "phone": ..., "email": ...}, ...]` |
| `def` 函数 | 每个功能封装为独立函数 |
| `while` 循环 | 主菜单循环，直到用户选择退出 |
| `if/elif/else` | 菜单分支、删除确认 |
| `for` 循环 + `enumerate` | 遍历列表并显示序号 |
| `f-string` | 格式化输出表格 |
| `str.lower()` / `strip()` | 字符串处理、输入清洗 |
| `in` 关键字 | 模糊搜索 |
| `list.pop()` / `list.append()` | 删除和添加联系人 |
| `json.load()` / `json.dump()` | 文件读写与数据持久化 |
| `try/except` | 处理文件不存在的异常 |
| `with` 语句 | 自动管理文件资源 |
| `if __name__ == "__main__"` | Python 程序入口惯例 |
