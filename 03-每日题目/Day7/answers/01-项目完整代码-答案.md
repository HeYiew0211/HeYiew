## 命令行通讯录管理器 — 完整代码

```python
"""
命令行通讯录管理器
====================
一个基于命令行的联系人管理工具，支持：
- 添加、查看、搜索、删除联系人
- 数据自动保存到 JSON 文件
- 异常处理和输入验证
"""

import json
import os

# 数据文件路径
DATA_FILE = "contacts.json"


# ============================================================
# 数据持久化函数
# ============================================================

def load_data():
    """
    从 JSON 文件加载通讯录数据。

    返回：
        list — 联系人列表，如果文件不存在或损坏则返回空列表
    """
    if not os.path.exists(DATA_FILE):
        print(f"数据文件 {DATA_FILE} 不存在，将创建新的通讯录。")
        return []

    try:
        with open(DATA_FILE, "r", encoding="utf-8") as f:
            contacts = json.load(f)
        # 确保加载的是列表类型
        if not isinstance(contacts, list):
            print("数据文件格式错误，将创建新的通讯录。")
            return []
        print(f"成功加载 {len(contacts)} 条联系人记录。")
        return contacts
    except (json.JSONDecodeError, IOError) as e:
        print(f"加载数据失败：{e}，将创建新的通讯录。")
        return []


def save_data(contacts):
    """
    将通讯录数据保存到 JSON 文件。

    参数：
        contacts: list — 联系人列表
    """
    try:
        with open(DATA_FILE, "w", encoding="utf-8") as f:
            json.dump(contacts, f, ensure_ascii=False, indent=2)
        print(f"成功保存 {len(contacts)} 条联系人记录到 {DATA_FILE}。")
    except IOError as e:
        print(f"保存数据失败：{e}")


# ============================================================
# 核心功能函数
# ============================================================

def add_contact(contacts):
    """
    添加新联系人。

    要求姓名和电话不能为空。

    参数：
        contacts: list — 联系人列表（原地修改）
    """
    print("\n----- 添加联系人 -----")
    name = input("请输入姓名：").strip()
    if not name:
        print("姓名不能为空，添加取消！")
        return

    phone = input("请输入电话：").strip()
    if not phone:
        print("电话不能为空，添加取消！")
        return

    email = input("请输入邮箱：").strip()
    # 邮箱可为空，空时填充默认值
    if not email:
        email = "无"

    contact = {
        "name": name,
        "phone": phone,
        "email": email,
    }
    contacts.append(contact)
    print(f"联系人「{name}」添加成功！")


def view_all(contacts):
    """
    以格式化表格显示所有联系人。

    参数：
        contacts: list — 联系人列表
    """
    print("\n----- 所有联系人 -----")
    if not contacts:
        print("通讯录为空，暂无联系人。")
        return

    # 打印表头
    print(f"{'序号':<6}{'姓名':<12}{'电话':<18}{'邮箱':<30}")
    print("-" * 66)

    # 遍历并格式化输出
    for index, contact in enumerate(contacts, start=1):
        print(
            f"{index:<6}"
            f"{contact['name']:<12}"
            f"{contact['phone']:<18}"
            f"{contact['email']:<30}"
        )
    print(f"\n共 {len(contacts)} 条记录。")


def search_contact(contacts):
    """
    按姓名模糊搜索联系人（不区分大小写）。

    参数：
        contacts: list — 联系人列表
    """
    print("\n----- 搜索联系人 -----")
    if not contacts:
        print("通讯录为空，暂无联系人可搜索。")
        return

    keyword = input("请输入要搜索的姓名关键词：").strip()
    if not keyword:
        print("搜索关键词不能为空！")
        return

    # 收集所有匹配结果（不区分大小写的模糊匹配）
    results = []
    for contact in contacts:
        if keyword.lower() in contact["name"].lower():
            results.append(contact)

    if not results:
        print(f"未找到姓名中包含「{keyword}」的联系人。")
        return

    # 打印搜索结果
    print(f"\n找到 {len(results)} 条匹配结果：")
    print(f"{'序号':<6}{'姓名':<12}{'电话':<18}{'邮箱':<30}")
    print("-" * 66)
    for index, contact in enumerate(results, start=1):
        print(
            f"{index:<6}"
            f"{contact['name']:<12}"
            f"{contact['phone']:<18}"
            f"{contact['email']:<30}"
        )


def delete_contact(contacts):
    """
    删除联系人。

    支持两种方式：
    1. 按序号删除 — 输入联系人序号直接删除
    2. 按姓名删除 — 模糊匹配姓名后删除（多匹配时让用户选择）

    参数：
        contacts: list — 联系人列表（原地修改）
    """
    print("\n----- 删除联系人 -----")
    if not contacts:
        print("通讯录为空，没有可删除的联系人。")
        return

    # 先显示所有联系人供用户参考
    view_all(contacts)

    choice = input("\n请输入要删除的联系人序号或姓名：").strip()
    if not choice:
        print("输入不能为空，删除取消！")
        return

    # ---------- 方式一：按序号删除 ----------
    if choice.isdigit():
        index = int(choice) - 1
        if 0 <= index < len(contacts):
            removed = contacts[index]
            confirm = input(f"确认删除「{removed['name']}」（{removed['phone']}）？(y/n)：").strip().lower()
            if confirm == "y":
                contacts.pop(index)
                print(f"联系人「{removed['name']}」已删除！")
            else:
                print("删除已取消。")
        else:
            print(f"序号 {choice} 无效，有效范围是 1~{len(contacts)}。")
        return

    # ---------- 方式二：按姓名删除 ----------
    matches = []
    for i, contact in enumerate(contacts):
        if choice.lower() in contact["name"].lower():
            matches.append((i, contact))

    if not matches:
        print(f"未找到姓名中包含「{choice}」的联系人。")
        return

    if len(matches) == 1:
        # 唯一匹配，确认后直接删除
        index, contact = matches[0]
        confirm = input(f"确认删除「{contact['name']}」（{contact['phone']}）？(y/n)：").strip().lower()
        if confirm == "y":
            contacts.pop(index)
            print(f"联系人「{contact['name']}」已删除！")
        else:
            print("删除已取消。")
    else:
        # 多个匹配，让用户选择具体删除哪一个
        print(f"\n找到 {len(matches)} 个匹配的联系人：")
        for j, (idx, c) in enumerate(matches, start=1):
            print(f"{j}. {c['name']} — {c['phone']} — {c['email']}")
        sub_choice = input("请输入要删除的序号（输入 0 取消）：").strip()
        if sub_choice.isdigit():
            sub_choice = int(sub_choice)
            if 1 <= sub_choice <= len(matches):
                idx, contact = matches[sub_choice - 1]
                confirm = input(f"确认删除「{contact['name']}」？(y/n)：").strip().lower()
                if confirm == "y":
                    contacts.pop(idx)
                    print(f"联系人「{contact['name']}」已删除！")
                else:
                    print("删除已取消。")
            else:
                print("序号无效，删除已取消。")
        else:
            print("输入无效，删除已取消。")


# ============================================================
# 主菜单
# ============================================================

def main():
    """主函数：程序入口，驱动菜单循环"""
    contacts = load_data()

    while True:
        # 打印菜单
        print("\n===== 命令行通讯录管理器 =====")
        print("1. 添加联系人")
        print("2. 查看所有联系人")
        print("3. 搜索联系人")
        print("4. 删除联系人")
        print("5. 退出")
        print("==============================")

        choice = input("请选择操作（1-5）：").strip()

        if choice == "1":
            add_contact(contacts)
        elif choice == "2":
            view_all(contacts)
        elif choice == "3":
            search_contact(contacts)
        elif choice == "4":
            delete_contact(contacts)
        elif choice == "5":
            save_data(contacts)
            print("通讯录数据已保存，再见！")
            break
        else:
            print("输入无效，请输入 1-5 之间的数字！")


if __name__ == "__main__":
    main()
```

## 知识点总结

| 知识点 | 在项目中的体现 |
|--------|----------------|
| **list** | `contacts` 列表存储所有联系人 |
| **dict** | 每个联系人是一个包含 `name`、`phone`、`email` 的字典 |
| **函数** | 将功能拆分为 6 个独立函数，每个函数单一职责 |
| **while 循环** | `while True` 实现菜单循环，用户选择退出时 `break` |
| **if/elif/else** | 菜单分支判断，根据用户输入执行不同操作 |
| **input() / print()** | 与用户交互的核心接口 |
| **f-string 格式化** | `f"{'姓名':<12}"` 实现对齐的表格输出 |
| **enumerate()** | `enumerate(contacts, start=1)` 同时获取索引和值 |
| **str 方法** | `.strip()` 去空格、`.lower()` 大小写不敏感、`.isdigit()` 判断数字 |
| **in 关键字** | `keyword in contact["name"]` 实现子串模糊匹配 |
| **json 模块** | `json.load()` / `json.dump()` 实现数据持久化 |
| **os.path.exists()** | 检查文件是否存在，避免 `FileNotFoundError` |
| **try/except** | 捕获文件读写异常和 JSON 解析异常 |
| **list.append()** | 向通讯录添加新联系人 |
| **list.pop(index)** | 按索引删除联系人 |
| **`__name__ == "__main__"`** | 脚本入口保护，确保被 import 时不自动执行 main() |
