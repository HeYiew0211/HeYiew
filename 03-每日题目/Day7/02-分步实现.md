先写出主程序骨架：一个 `while True` 循环 + 菜单打印 + 用户输入 + 分支判断。

```python
def main():

        choice = input("请选择操作（1-5）：").strip()

        if choice == "1":
            print(">>> 添加联系人功能（待实现）")
        elif choice == "2":
            print(">>> 查看所有联系人功能（待实现）")
        elif choice == "3":
            print(">>> 搜索联系人功能（待实现）")
        elif choice == "4":
            print(">>> 删除联系人功能（待实现）")
```

**解释：**

- `if/elif/else` 根据用户输入分发到不同功能
- 先把框架跑通，再逐个实现功能函数

## 第 2 步：实现 load_data 和 save_data

数据持久化是通讯录的基础，先实现它。

```python
import json
import os

# 定义 JSON 文件的保存路径
DATA_FILE = "contacts.json"


def load_data():
    """
    从 JSON 文件加载通讯录数据。

    返回：
        list — 联系人列表，如果文件不存在则返回空列表
    """
    # 检查文件是否存在
    if not os.path.exists(DATA_FILE):
        print(f"数据文件 {DATA_FILE} 不存在，将创建新的通讯录。")
        return []

    try:
        with open(DATA_FILE, "r", encoding="utf-8") as f:
            contacts = json.load(f)
        print(f"成功加载 {len(contacts)} 条联系人记录。")
        return contacts
    except (json.JSONDecodeError, IOError) as e:
        print(f"加载数据失败：{e}，将创建新的通讯录。")
        return []


def save_data(contacts):
    """
    将通讯录数据保存到 JSON 文件。

    参数：
        contacts: list — 联系人列表
    """
    try:
        with open(DATA_FILE, "w", encoding="utf-8") as f:
            json.dump(contacts, f, ensure_ascii=False, indent=2)
        print(f"成功保存 {len(contacts)} 条联系人记录到 {DATA_FILE}。")
    except IOError as e:
        print(f"保存数据失败：{e}")
```

**解释：**
- `os.path.exists()` 先检查文件是否存在，避免无意义的异常
- `json.load(f)` 将 JSON 文件内容解析为 Python 对象（这里是 list of dict）
- `json.dump(obj, f, ensure_ascii=False, indent=2)` 格式化写入，`ensure_ascii=False` 保证中文正常显示
- 外层用 `try/except` 捕获文件损坏或权限问题等异常

在主菜单中集成：启动时调用 `load_data()`，退出时调用 `save_data()`。

```python
def main():
    contacts = load_data()  # 启动时加载

    while True:
        # ... 菜单代码保持不变 ...
        elif choice == "5":
            save_data(contacts)  # 保存后退出
            print("再见！")
            break
```

## 第 3 步：实现 add_contact 和 view_all

### add_contact

```python
def add_contact(contacts):
    """
    添加新联系人。

    参数：
        contacts: list — 联系人列表（会被原地修改）
    """
    print("\n----- 添加联系人 -----")
    name = input("请输入姓名：").strip()
    if not name:
        print("姓名不能为空，添加取消！")
        return

    phone = input("请输入电话：").strip()
    if not phone:
        print("电话不能为空，添加取消！")
        return

    email = input("请输入邮箱：").strip()

    # 构造联系人 dict 并追加到列表
    contact = {
        "name": name,
        "phone": phone,
        "email": email,
    }
    contacts.append(contact)
    print(f"联系人「{name}」添加成功！")
```

### view_all

```python
def view_all(contacts):
    """
    显示所有联系人。

    参数：
        contacts: list — 联系人列表
    """
    print("\n----- 所有联系人 -----")
    if not contacts:
        print("通讯录为空，暂无联系人。")
        return

    # 打印表头
    print(f"{'序号':<6}{'姓名':<12}{'电话':<16}{'邮箱':<25}")
    print("-" * 59)

    # 遍历联系人，格式化打印
    for index, contact in enumerate(contacts, start=1):
        print(
            f"{index:<6}"
            f"{contact['name']:<12}"
            f"{contact['phone']:<16}"
            f"{contact['email']:<25}"
        )
    print(f"\n共 {len(contacts)} 条记录。")
```

**解释：**
- `input().strip()` 去除两端空格；空姓名或空电话时 `return` 取消操作
- `enumerate(contacts, start=1)` 同时获取索引和元素，序号从 1 开始
- f-string 格式化：`<6` 表示左对齐占 6 个字符宽度，`<12` 占 12 个宽度
- `contacts.append(contact)` 向列表末尾添加元素

## 第 4 步：实现 search_contact

```python
def search_contact(contacts):
    """
    按姓名模糊搜索联系人。

    参数：
        contacts: list — 联系人列表
    """
    print("\n----- 搜索联系人 -----")
    if not contacts:
        print("通讯录为空，暂无联系人。")
        return

    keyword = input("请输入要搜索的姓名关键词：").strip()
    if not keyword:
        print("搜索关键词不能为空！")
        return

    # 遍历并收集匹配结果
    results = []
    for contact in contacts:
        if keyword.lower() in contact["name"].lower():
            results.append(contact)

    if not results:
        print(f"未找到姓名中包含「{keyword}」的联系人。")
        return

    print(f"\n找到 {len(results)} 条匹配结果：")
    print(f"{'序号':<6}{'姓名':<12}{'电话':<16}{'邮箱':<25}")
    print("-" * 59)
    for index, contact in enumerate(results, start=1):
        print(
            f"{index:<6}"
            f"{contact['name']:<12}"
            f"{contact['phone']:<16}"
            f"{contact['email']:<25}"
        )
```

**解释：**
- `keyword.lower() in contact["name"].lower()` 做不区分大小写的模糊匹配
- `in` 关键字判断子串是否存在，使搜索更灵活（如输入"张"可以匹配"张三"、"张伟"等）
- 先收集匹配结果到 `results` 列表，再统一输出

## 第 5 步：实现 delete_contact

```python
def delete_contact(contacts):
    """
    删除联系人。

    支持两种删除方式：
    1. 按序号删除（直接输入数字）
    2. 按姓名删除（输入姓名后，若有多个匹配则列出让用户选择）

    参数：
        contacts: list — 联系人列表（会被原地修改）
    """
    print("\n----- 删除联系人 -----")
    if not contacts:
        print("通讯录为空，没有可删除的联系人。")
        return

    # 先显示现有联系人
    view_all(contacts)

    choice = input("\n请输入要删除的联系人序号或姓名：").strip()
    if not choice:
        print("输入不能为空，删除取消！")
        return

    # 方式 1：按序号删除
    if choice.isdigit():
        index = int(choice) - 1
        if 0 <= index < len(contacts):
            removed = contacts.pop(index)
            print(f"联系人「{removed['name']}」已删除！")
        else:
            print(f"序号 {choice} 无效，请检查！")
        return

    # 方式 2：按姓名删除
    matches = []
    for i, contact in enumerate(contacts):
        if choice.lower() in contact["name"].lower():
            matches.append((i, contact))

    if not matches:
        print(f"未找到姓名中包含「{choice}」的联系人。")
        return

    if len(matches) == 1:
        # 唯一匹配，直接删除
        index, contact = matches[0]
        confirm = input(f"确认删除「{contact['name']}」（{contact['phone']}）？(y/n)：").strip().lower()
        if confirm == "y":
            contacts.pop(index)
            print(f"联系人「{contact['name']}」已删除！")
        else:
            print("删除已取消。")
    else:
        # 多个匹配，让用户选择
        print(f"\n找到 {len(matches)} 个匹配的联系人：")
        for j, (idx, c) in enumerate(matches, start=1):
            print(f"{j}. {c['name']} — {c['phone']}")
        sub_choice = input("请输入要删除的序号（输入 0 取消）：").strip()
        if sub_choice.isdigit():
            sub_choice = int(sub_choice)
            if 1 <= sub_choice <= len(matches):
                idx, contact = matches[sub_choice - 1]
                contacts.pop(idx)
                print(f"联系人「{contact['name']}」已删除！")
            else:
                print("删除已取消。")
        else:
            print("删除已取消。")
```

**解释：**
- `choice.isdigit()` 判断输入是否是纯数字，以此区分"按序号删除"和"按姓名删除"
- `list.pop(index)` 按索引删除元素并返回被删除的元素
- 当姓名匹配到多个结果时，让用户二次选择避免误删
- 删除前需要用户确认（`y/n`），防止误操作
